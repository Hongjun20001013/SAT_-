% Slide 3: What is a Ratio?
\begin{frame}{What is a Ratio?}
    A \textbf{ratio} is a comparison between two numbers. It can be written in three different ways:
    
    \[
    \frac{a}{b}, \quad a:b, \quad \text{or } a \text{ to } b
    \]

    \textbf{Example:} The ratio of boys to girls in a class is 3 to 4.  
    - This means for every 3 boys, there are 4 girls.  
    - The fraction form is \( \frac{3}{4} \).
\end{frame}

% Slide 4: Rates and Unit Rates
\begin{frame}{Rates and Unit Rates}
    A \textbf{rate} is a special type of ratio that compares different units.

    \textbf{Example:} A car travels 180 miles in 3 hours. The rate is:

    \[
    \frac{180 \text{ miles}}{3 \text{ hours}} = 60 \text{ miles per hour}
    \]

    A \textbf{unit rate} simplifies the rate so that the denominator is 1.

    \textbf{Example:} If 12 apples cost \$6, the unit price per apple is:

    \[
    \frac{6 \text{ dollars}}{12 \text{ apples}} = 0.50 \text{ dollars per apple}
    \]
\end{frame}

\begin{frame}{Unit Conversion}

\textbf{Key Idea:} Multiply by a conversion factor equal to 1.

\[
\text{Quantity in old unit} \times \frac{\text{new unit}}{\text{old unit}} = \text{Quantity in new unit}
\]

\vspace{0.5em}
\textbf{Example 1: Inches to Feet}
\[
36 \text{ inches} \times \frac{\frac{1}{12} \text{ foot}}{{} \text{ 1 inches}} = 3 \text{ feet}
\]

\vspace{0.5em}
\textbf{Example 2: Minutes to Seconds}
\[
5 \text{ minutes} \times \frac{60 \text{ seconds}}{1 \text{ minute}} = 300 \text{ seconds}
\]

\vspace{1em}
\textbf{Tips:}
\begin{itemize}
    \item Always write units clearly.
    \item Cancel the units you don’t need.
    \item Use a conversion factor that equals 1.
    \item Check your final units!
\end{itemize}

\end{frame}



\begin{frame}{Example: Unit Conversion}

\textbf{Question:} A certain town has an area of 4.36 square miles. What is the area, in \textbf{square yards}, of this town? \\

\textbf{Given:} \(1 \text{ mile} = 1,760 \text{ yards}\) \\

\begin{enumerate}[(A)]
    \item 404
    \item 7,674
    \item 710,459
    \item 13,505,536
\end{enumerate}

\end{frame}







\begin{frame}{Solution:Unit Conversion}

\textbf{Step 1: Understanding the conversion factor} \\
We know that:
\[
1 \text{ mile} = 1,760 \text{ yards}
\]
Since we are dealing with square miles, we square both sides:
\[
1 \text{ square mile} = (1,760 \text{ yards})^2 = 3,097,600 \text{ square yards}
\]

\textbf{Step 2: Multiply by given area} \\
The given area is \( 4.36 \) square miles. Converting to square yards:
\[
4.36 \times 3,097,600 = 13,505,536 \text{ square yards}
\]

\textbf{Final Answer:} \\
\[
\boxed{13,505,536} \quad \text{(Option D)}
\]

\end{frame}

% Slide 5: Solving Proportions
\begin{frame}{Solving Proportions}
    A \textbf{proportion} is an equation stating that two ratios are equal:

    \[
    \frac{a}{b} = \frac{c}{d}
    \]

    \textbf{Cross Multiplication:} If \( \frac{3}{4} = \frac{x}{12} \), solve for \( x \):

    \[
    3 \times 12 = 4 \times x
    \]

    \[
    36 = 4x \quad \Rightarrow \quad x = 9
    \]
\end{frame}


\begin{frame}{Question: Solving for \( n \)}
\textbf{Problem Statement:} 

If 

\[
\frac{4a}{b} = 6.7 \quad \text{and} \quad \frac{a}{bn} = 26.8,
\]

what is the value of \( n \)?

\end{frame}

\begin{frame}{Solution: Solving for \( n \)}

\textbf{Step 1: Express \( a/b \) in terms of known values} \\
From the first equation:
\[
\frac{4a}{b} = 6.7
\]
Dividing both sides by 4:
\[
\frac{a}{b} = \frac{6.7}{4} = 1.675
\]

\textbf{Step 2: Substitute into the second equation} \\
\[
\frac{a}{bn} = 26.8
\]
Rewriting using \( \frac{a}{b} = 1.675 \):
\[
\frac{1.675}{n} = 26.8
\]



\textbf{Final Answer:} \\
\[
\boxed{0.0625}
\]

\end{frame}

\begin{frame}{Ratio Word Problem}
The ratio of blueberries to grapes in a fruit salad is $5$ to $12$. If there are $x$ blueberries in the fruit salad, which expression represents the number of grapes in the fruit salad?
\begin{itemize}
    \item[(A)] $\dfrac{5}{12}x$
    \item[(B)] $\dfrac{12}{5}x$
    \item[(C)] $\dfrac{12}{17}x$
    \item[(D)] $60x$
\end{itemize}
\end{frame}


\begin{frame}{Answer}
To maintain the ratio $5:12$ (blueberries to grapes), we set up the proportion:
\[
\dfrac{5}{12} = \dfrac{x}{g} \Rightarrow g = \dfrac{12}{5}x
\]
\textbf{Correct answer:} (B) $\dfrac{12}{5}x$
\end{frame}

\begin{frame}{Unit Conversion: Area}
The area of Wompatuck State Park in Massachusetts is $5.5$ square miles. Which of the following is closest to the area of this park, in \textbf{square kilometers}? (Use $1$ kilometer $= 0.62$ mile.)
\begin{itemize}
    \item[(A)] $2.1$
    \item[(B)] $3.4$
    \item[(C)] $8.9$
    \item[(D)] $14.3$
\end{itemize}
\end{frame}


\begin{frame}{Answer}
We are given:
\[
\text{Area in square miles} = 5.5, \quad 1 \text{ kilometer} = 0.62 \text{ mile}
\]
So,
\[
1 \text{ square mile} = \left(\dfrac{1}{0.62}\right)^2 \approx 2.6 \text{ square kilometers}
\]
\[
5.5 \text{ square miles} \approx 5.5 \times 2.6 = 14.3 \text{ square kilometers}
\]
\textbf{Correct answer:} (D) $14.3$
\end{frame}


\begin{frame}{Scale Modeling}
A model of a building has a scale where a width of $5$ inches in the model represents an actual width of $16$ feet. A room in the building has an actual width of $10w$ feet. Which expression represents the width, in inches, of the room in the model?
\begin{itemize}
    \item[(A)] $\dfrac{5}{8}w$
    \item[(B)] $\dfrac{25}{8}w$
    \item[(C)] $8w$
    \item[(D)] $32w$
\end{itemize}
\end{frame}


\begin{frame}{Answer}
The scale is $5$ inches in the model : $16$ feet in reality.

Let the actual width be $10w$ feet. To find the width in the model, apply the ratio:
\[
\frac{5}{16} = \frac{\text{model width}}{10w} \Rightarrow \text{model width} = \frac{5}{16} \cdot 10w = \frac{50}{16}w = \frac{25}{8}w
\]
\textbf{Correct answer:} (B) $\dfrac{25}{8}w$
\end{frame}
